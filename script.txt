const createBtn = document.querySelector("#create");
const formOverlay = document.querySelector(".form");
const form = document.querySelector("form");
const formh1 = document.querySelector(".formh1");
const closeBtn = document.querySelector("#close");
const products = document.querySelector(".products");

// Linking productArr to Local Storage, & setting a condition to return a BLANK ARRAY
// if the productArr is empty, insted of setting it to NULL.
const productArr = JSON.parse(localStorage.getItem("items")) || [];
let updateIndex = null;

createBtn.addEventListener("click", () => {
    formOverlay.style.display = "flex";
    formh1.textContent = "Create Product";
    form[4].textContent = "Create";
});

closeBtn.addEventListener("click", () => {
    formOverlay.style.display = "none";
});

let ui = () =>{
    products.innerHTML = "";
    productArr.forEach((elem, index) => {
        products.innerHTML += `<div class="product-card">
                <div class="img">
                    <img src="${elem.image}" alt="Product-img">
                </div>
                
                <div class="text">
                    <h3>${elem.productName}</h3>
                    <p>${elem.description}</p>
                    <p>${elem.price}</p>
                </div>
    
                <div class="btns">
                    <button onClick="updateProduct('${elem.productName}')" id="update">Update</button>
                    <button onClick="deleteProduct(${index})" id="delete">Delete</button>
                </div>
            </div>`
    });
};

ui();

form.addEventListener("submit", (event, index) => {
    event.preventDefault();
    
    let productName = event.target[0].value;
    let description = event.target[1].value;
    let price = event.target[2].value;
    let image = event.target[3].value;

    if(productName.trim() === "" || description.trim() === "" || price.trim() === "" || image === ""){
        alert("Please fill all the fields!");
        return;
    }

    let obj = {
        id: Date.now(),
        productName,
        description,
        price,
        image,
    };

    if(updateIndex !== null){
        productArr[updateIndex] = obj;
        updateIndex = null;
        localStorage.setItem("items", JSON.stringify(productArr));  // Saving data into the browser's local storage
    }else{
        productArr.push(obj);
        localStorage.setItem("items", JSON.stringify(productArr));
    }

    ui();
    
    form.reset();
    formOverlay.style.display = "none";
});

const updateProduct = (name) => {
    formOverlay.style.display = "flex";
    formh1.textContent = "Update Product";
    form[4].textContent = "Update";
    let product = productArr.find((elem) => elem.productName === name);
    updateIndex = productArr.findIndex((elem) => elem.productName === name);

    form[0].value = product.productName
    form[1].value = product.description
    form[2].value = product.price
    form[3].value = product.image
};

const deleteProduct = (index) => {
    productArr.splice(index, 1);
    localStorage.setItem("items", JSON.stringify(productArr));
    ui();
};
